Blueberry Group : BBWT3 - Tree

Created by Jason Cozza on Sep 21, 2020

Overview

This page demonstrates how to insert tree's into your BBWT3 code.


Example:

Example HTML:

<div class="ui-g">
    <div class="ui-g-12">
        <label for="tree">Label</label>
    </div>
 
    <div class="ui-g-12">
        <button type="button" pButton (click)="root()" class="btn-standard-size"
                   label="Root">
                <i class="material-icons">folder</i>
        </button>
        <p-tree id="tree" [value]="treeData" selectionMode="single"
                   [(selection)]="selectedFile"
                   (onNodeCollapse)="nodeCollapse($event)"
                   (onNodeExpand)="nodeExpand($event)"></p-tree>
    </div>
    <div class="ui-g-12">
        <div style="margin-top:8px">Selected Node: {{selectedFile ? selectedFile.label :
            'none' } }
        </div>
    </div>
    <div class="ui-g-12">
        <button type="button" pButton (click)="addItem()" class="btn-standard-size"
                   label="Add">
                    <i class="material-icons">add</i>
        </button>
        <button type="button" pButton (click)="editItem()"class="btn-standard-size"
                   label="Edit" [disabled]="selectedFile == null">
                    <i class="material-icons">edit</i>
        </button>
        <button type="button" pButton (click)="deleteItem(selectedFile)"
                   class="btn-standard-size" label="Delete"
                   [disabled]="selectedFile == null">
                    <i class="material-icons">delete_forever</i>
        </button>
    </div>
</div>

Example Typescript:

export class TreeComponent implements OnInit {
 
editingMode: boolean = false;
addFormVisible: boolean = false;
selectedFile: File = null;
newItem: File;
treeData: any;
itemTypes: Array
<SelectItem>;
    selectedType: number;
    constructor(private fileSvc: FileService) { }
 
    ngOnInit() {
    this.resetForm()
 
    this.itemTypes = [
    { label: 'Folder', value: 1 },
    { label: 'File', value: 2 }
    ];
    this.selectedType = 1
    this.loadItems()
    }
 
    loadItems() {
    this.fileSvc.getAll()
    .then((result) => {
    this.treeData = result;
    });
    }
 
    addItem() {
    this.addFormVisible = true;
    this.editingMode = false;
    this.selectedType = 1
    }
 
    editItem() {
    this.editingMode = true;
    this.newItem.data = this.selectedFile.data
    this.newItem.label = this.selectedFile.label
 
    this.selectedType = this.selectedFile.type
    this.addFormVisible = true;
    }
 
    deleteItem(item: any) {
    this.fileSvc.delete(item.id)
    .then((result) => {
    this.loadItems()
    });
    this.root()
    }
 
    commit() {
    this.newItem.type = this.selectedType;
 
    if (!this.editingMode) {
    if (this.selectedFile != null) {
    this.newItem.parentId = this.selectedFile.id;
    if (this.selectedFile.type == 1) {
    this.fileSvc.create(this.newItem)
    .then((result) => {
    this.loadItems()
    });
    } else {
    this.newItem.parentId = this.selectedFile.parentId;
    this.fileSvc.create(this.newItem)
    .then((result) => {
    this.loadItems()
    });
    }
    } else {
    this.fileSvc.create(this.newItem)
    .then((result) => {
    this.loadItems()
    });
    }
    } else {
    this.newItem.id = this.selectedFile.id
    this.fileSvc.update(this.newItem)
    .then((result) => {
    this.loadItems()
    });
    }
 
    this.resetForm()
    this.addFormVisible = false
    }
 
    finishEdit() {
    this.addFormVisible = false;
    }
 
    saveNodeState(event) {
    let item: File
    item = {
    id: event.node.id,
    id_original: event.node.id_original,
    parentId: "",
    parentId_original: 0,
    parent: null,
    label: event.node.label,
    data: event.node.data,
    type: 0,
    icon: "",
    expandedIcon: "",
    collapsedIcon: "",
    expanded: !event.node.expanded,
    children: []
    }
 
    this.fileSvc.updateNoMsg(item)
    .subscribe(data => {
    })
    }
 
    nodeExpand(event) {
    this.saveNodeState(event)
    }
 
    resetForm() {
    this.newItem = {
    id: "",
    id_original: 0,
    parentId: "",
    parentId_original: 0,
    parent: null,
    label: "",
    data: "",
    type: 0,
    icon: "",
    expandedIcon: "",
    collapsedIcon: "",
    expanded: false,
    children: []
    }
    this.selectedFile = null
    }
 
    root() {
    this.selectedFile = null
    }
 
    }
