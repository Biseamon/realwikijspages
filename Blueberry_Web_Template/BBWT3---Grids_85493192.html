Blueberry Group : BBWT3 - Grids

Created by Jason Cozza on Sep 21, 2020

Overview

This page details how to compose grids within BBWT3.


Content of the grid should be visible without horizontal scrolling. But you can always discuss it with your manager.

Example Typescript:

export class GridsComponent implements OnInit {
 
public filterCommand: IFilterCommand;
public filter: { customerCode: string, id: number, orderDate: Date, requiredDate: Date, shippedDate: Date };
 
    selectedOrders: Order[];
 
    page: PageResult<Order> = {
        items: [],
        total: 0
    };
 
    constructor(private orderService: OrderService) { }
 
    ngOnInit() {
        this.selectedOrders = [];
        this.initDefaultFilterCommand()
        this.load(this.filterCommand)
    }
 
    onShowSelected() {
        var str: string = '';
        this.selectedOrders.forEach(item => { str += item.id.toString() + ';' });
    }
 
    load(filterCommand: IFilterCommand) {
        this.orderService.getPage(filterCommand)
            .then((result: PageResult<Order>) => {
                if (result) {
                    this.page.items = result.items;
                    this.page.total = result.total;
                }
            });
    }
 
    onGridLazyLoad(event: LazyLoadEvent) {
        // map grid lazy load event to filter
        this.filterCommand = new FilterCommand(event);
        this.fillFilters();
 
        //load data
        this.load(this.filterCommand);
    }
 
    clearFilter() {
        this.initDefaultFilterCommand();
        this.load(this.filterCommand);
    }
 
    search() {
        this.fillFilters();
        this.load(this.filterCommand);
    }
 
    private initDefaultFilterCommand() {
        this.filterCommand = {
            first: 0,
            rows: 10,
            sortOrder: 1,
            filters: []
        };
        this.filter = {
            customerCode: null,
            id: null,
            orderDate: null,
            requiredDate: null,
            shippedDate: null
        };
    }
 
    private fillFilters() {
        this.filterCommand.filters = [];
 
        if (this.filter.id) {
            this.filterCommand.filters.push(new NumberFilter('id', this.filter.id));
        }
        if (this.filter.customerCode) {
            this.filterCommand.filters.push(new StringFilter('CustomerCode', this.filter.customerCode));
        }
        if (this.filter.orderDate) {
            this.filterCommand.filters.push(new DateFilter('OrderDate', this.filter.orderDate));
        }
        if (this.filter.shippedDate) {
            this.filterCommand.filters.push(new DateFilter('ShippedDate', this.filter.shippedDate));
        }
        if (this.filter.requiredDate) {
            this.filterCommand.filters.push(new DateFilter('RequiredDate', this.filter.requiredDate));
        }
    }
}
